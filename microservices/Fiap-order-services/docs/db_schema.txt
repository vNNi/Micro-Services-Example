Project social_blogging_site {
database_type: 'Mysql'
Note: 'Uebaaaaa'
}

Table products {
id integer [primary key]
title varchar
description varchar
sort varchar
price float
category varchar
visited int
created_at timestamp
}

Table user {
id integer [primary key]
name varchar
role role
favorites integer [ref: > products.id]
created_at timestamp
}

table ticket {
id integer [primary key]
user_id []
title varchar
description varchar
risk risk
}

Enum risk {
  low
  medium
  high
}

Enum role {
  admin
  client
  seller
}

Table order {
  id integer [primary key]
  product_id integer
  user_id integer
  shipping_estimate integer
  product_price float
  title varchar
  description varchar
  completed boolean
}

Ref: order.product_id > products.id

Ref: order.user_id > user.id

Ref: user.id > ticket.id
